58  inline void Lock()
59  {
60  if (!m_bLocked)

62  xnOSEnterCriticalSection(&m_hCS);

64  }

66 
67  inline void Unlock()

71  xnOSLeaveCriticalSection(&m_hCS);
72  m_bLocked = FALSE;
73  }
74  }
75

78  XnBool m_bLocked;
